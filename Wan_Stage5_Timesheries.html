def create_time_series(X, y, time_steps):
    X_series, y_series = [], []
    for i in range(len(X) - time_steps):
        X_series.append(X[i:i + time_steps])
        y_series.append(y[i + time_steps])
    return np.array(X_series), np.array(y_series)
